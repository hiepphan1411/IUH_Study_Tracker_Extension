<script>
    function toggleDropdown(button) {
        var dropdownContent = button.parentElement.querySelector('.dropdown-content-thaotac');
        var allDropdowns = document.querySelectorAll('.dropdown-content-thaotac');
        allDropdowns.forEach(function (dropdown) {
            if (dropdown !== dropdownContent) {
                dropdown.style.display = 'none';
            }
        });
        dropdownContent.style.display = dropdownContent.style.display === 'flex' ? 'none' : 'flex';

    }
    document.addEventListener('click', function (event) {
        var dropdowns = document.getElementsByClassName('dropdown-content-thaotac');
        var buttons = document.getElementsByClassName('dropbtn-thaotac');
        var targetElement = event.target;
        var clickedInsideDropdown = false;
        for (var i = 0; i < dropdowns.length; i++) {
            if (dropdowns[i].contains(targetElement) || buttons[i].contains(targetElement)) {
                clickedInsideDropdown = true;
                break;
            }
        }
        if (!clickedInsideDropdown) {
            for (var i = 0; i < dropdowns.length; i++) {
                dropdowns[i].style.display = 'none';
            }
        }
    });

</script>

        });
    }
